# 配置开发环境-安装MySQL

# 安装MySQL 5.7

# 1. 下载地址

官网下载地址 (opens new window)

wget https://downloads.mysql.com/archives/get/p/23/file/mysql-5.7.32-linux-glibc2.12-x86_64.tar.gz
1

# 2. 检查工作

  1. 检查当前系统是否安装过mysql
rpm -qa|grep mariadb
1

如果存在如下：

请执行以下卸载命令：

rpm -e --nodeps  mariadb-libs
1

删除etc目录下的my.cnf ，一定要删掉，等下再重新建，之前我将就用这个文件，后面改配置各种不生效，因为lz此目录下没有my.cnf文件，没有执行可命令

rm /etc/my.cnf
1
  1. 检查当前mysql依赖环境

执行安装命令前，先执行查询命令

rpm -qa|grep libaio
1

rpm -qa|grep net-tools

# 3. 检查有无安装过mysql 用户组，没有的话创建

  • 检查mysql 用户组是否存在

    cat /etc/group | grep mysql
cat /etc/passwd |grep mysql
    1
    2

  • 创建mysql 用户组和用户

    groupadd mysql
useradd -r -g mysql mysql
    1
    2

# 4. 安装

  1. 通过 Xftp工具把在下载好文件发送到Linux 的opt目录下，或者wget下载
wget https://downloads.mysql.com/archives/get/p/23/file/mysql-5.7.32-linux-glibc2.12-x86_64.tar.gz
1
  1. 解压安装mysql安装包到
tar -zxvf mysql-5.7.32-linux-glibc2.12-x86_64.tar.gz -C  /opt/develop/mysql
1
  1. 修改名称为mysql5.7
mv mysql-5.7.32-linux-glibc2.12-x86_64/ mysql5.7
1
  1. 进入/opt/develop/mysql/mysql5.7/bin/目录，编译安装并初始化mysql,务必记住数据库管理员临时密码,如下图画红色框的部分。
./mysqld --initialize --user=mysql --datadir=/opt/develop/mysql/mysql5.7/data --basedir=/opt/develop/mysql/mysql5.7 # 添加用户组
1

一定要记住临时密码：g(M4vy:zodRs _Vk:IWWUt1DR

(81条消息) MySQL5.7报错–initialize specified but the data directory has files in it. Aborting._鹏老板的博客-CSDN博客 (opens new window)

  1. 编写etc目录下的my.cnf 配置文件，并添加配置
//进入配置文件
vim /etc/my.cnf
1
2

配置文件内容

[client]
port = 3306

#character-set-server=utf8
 
[mysqld]
#设置3306端口
port = 3306
# 设置mysql的安装目录 tips:这里的目录是你自己的安装目录，这个是我的安装目录，你不能用的哦
basedir=/opt/develop/mysql/mysql5.7
# 设置mysql数据库的数据的存放目录 tips:同上一条
datadir=/opt/develop/mysql/mysql5.7/data
# 允许最大连接数
max_connections=200
# 服务端使用的字符集默认为8比特编码的latin1字符集
character-set-server=utf8
collation-server=utf8_general_ci
# 创建新表时将使用的默认存储引擎
default-storage-engine=INNODB
# MySQL5.7版本sql_mode=only_full_group_by问题解决办法
sql_mode=STRICT_TRANS_TABLES,NO_ZERO_IN_DATE,NO_ZERO_DATE,ERROR_FOR_DIVISION_BY_ZERO,NO_AUTO_CREATE_USER,NO_ENGINE_SUBSTITUTIONls


#表名大小写不明感，敏感为
lower_case_table_names=1

# 解决MySQL Unknown table engine 'FEDERATED''
federated

# skip-grant-tables
 
[mysql]
# 设置mysql客户端默认字符集
default-character-set=utf8

# 5. 启动mysql服务

  1. 查询服务
ps -ef|grep mysql
ps -ef|grep mysqld
1
2

  1. 启动服务
/opt/develop/mysql/mysql5.7/support-files/mysql.server start
1

启动问题：

启动mysql报错The server quit without updating PID file！ - 腾讯云开发者社区-腾讯云 (tencent.com) (opens new window)

Please read "Security" section of the manual to find out how to run mysqld as root! (opens new window)

  1. 添加软连接，并重启mysql 服务 😄
# 添加软连接
ln -s /opt/develop/mysql/mysql5.7/support-files/mysql.server /etc/init.d/mysql
ln -s /opt/develop/mysql/mysql5.7/bin/mysql /usr/bin/mysql
# 重启mysql服务
service mysql restart

# 6. 设置开机自启

# 将服务文件拷贝到init.d下，并重命名为mysql
cp /opt/develop/mysql/mysql5.7/support-files/mysql.server /etc/init.d/mysqld
# 赋予可执行权限
chmod +x /etc/init.d/mysqld
# 添加服务
chkconfig --add mysqld
# 显示服务列表
chkconfig --list
