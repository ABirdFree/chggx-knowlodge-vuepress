# 2. 常用命令

1. docker ps # 列出所有在运行的容器信息
2.docker images # 查看镜像 
2. docker stop 容器名  #停止nexus
3. docker start 容器名 #启动nexus 启动时间大约1分钟
4. docker restart 容器名 # 重启容器mysql
5. docker stop 容器名/容器id  # 停止容器
6. docker rm 容器名/容器id # 删除容器
7. docker rmi xxxx 删除镜像

 

1. docker inspect 容器id # 查询容器信息
2. docker exct -it 容器id/容器名 bash  #(进去容器内部)
3. Exit #退出
docker version # 查看版本
systemctl start docker # 启动docker
systemctl stop docker # 关闭停止docker
systemctl daemon-reload # 重新加载并配置(修改文件后)
systemctl restart docker # 重新加载并配置(修改文件后)

# 5. Docker安装Nginx

1. 拉取
docker pull nginx

2. 创建容器
 docker run -d --name=nginx -p 8080:80 nginx
 
 docker run -d -p 8080:80 -v $PWD/dist:/usr/share/nginx/html nginx
 
 配置文件位置：/etc/nginx/
 
docker run --name nginx-index -d -p 80:80 -v $PWD/dist:/usr/share/nginx/html -v /opt/nginx/nginx.conf:/etc/nginx/nginx.conf:ro -v /opt/nginx/conf.d:/etc/nginx/conf.d nginx

[root@VM_0_2_centos opt]# docker run --name nginx-index -d -p 80:80 -v $PWD/dist/:/usr/share/nginx/html -v /opt/nginx/nginx.conf:/etc/nginx/nginx.conf -v /opt/nginx/conf.d/default.conf:/etc/nginx/conf.d/default.conf nginx

docker run --name nginx-index -d -p 80:80 -v /opt/blog/index/dist/:/usr/share/nginx/html -v /opt/nginx/nginx.conf:/etc/nginx/nginx.conf nginx

# 6. docker部署yapi

YApi的私有云部署

# 1. 启动YApi-mongo：

docker run -d --name mongo-yapi mongo
1

# 2. 初始化YApi:

docker run -it --rm \
--link mongo-yapi:mongo \
--entrypoint npm \
--workdir /api/vendors \
 registry.cn-hangzhou.aliyuncs.com/anoy/yapi \
 run install-server

# 3. 启动YApi：

docker run -d \
  --name yapi \
  --link mongo-yapi:mongo \
  --workdir /api/vendors \
  -p 3000:3000 \
  registry.cn-hangzhou.aliyuncs.com/anoy/yapi \
  server/app.js
